\notebookpage{fill dotted,paginate}

\notebookpage{
  box={A full width box},
  todo=2,

}

\pgfcalendarcalc{\year-\month-\day}{\ThisDay}

  },

  calendar/gone/.style={},
  calendar/weekend/.style={green!50!black,font=\bfseries},
  calendar={[bottom,fill=green!10]{\NextMonth}},
  paginate,
}
